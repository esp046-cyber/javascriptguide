5Debounce & Throttle
Control function execution rate for performance optimization. 
function debounce(func, delay) {
    let timeout;
    return (...args) => {
        clearTimeout(timeout);
        timeout = setTimeout(() => {
            func(...args);
        }, delay);
    };
}

const search = debounce((query) => {
    console.log('Searching:', query);
}, 300);
6Memoization
Cache expensive function results for better performance. 
function memoize(fn) {
    const cache = new Map();
    return (...args) => {
        const key = JSON.stringify(args);
        if (cache.has(key)) {
            return cache.get(key);
        }
        const result = fn(...args);
        cache.set(key, result);
        return result;
    };
}

// Todo List Application
const todos = [];
let filter = 'all';

function addTodo(text) {
    if (!text.trim()) return;
    
    const todo = {
        id: Date.now(),
        text: text,
        completed: false
    };
    
    todos.push(todo);
    renderTodos();
}

function deleteTodo(id) {
    const index = todos.findIndex(t => t.id === id);
    if (index > -1) {
        todos.splice(index, 1);
        renderTodos();
    }
}

function toggleTodo(id) {
    const todo = todos.find(t => t.id === id);
    if (todo) {
        todo.completed = !todo.completed;
        renderTodos();
    }
}

function getFilteredTodos() {
    if (filter === 'active') {
        return todos.filter(t => !t.completed);
    } else if (filter === 'completed') {
        return todos.filter(t => t.completed);
    }
    return todos;
}

function renderTodos() {
    const container = document.getElementById('todo-list');
    const filtered = getFilteredTodos();
    
    container.innerHTML = filtered.map(todo => \`
        <div class="todo-item">
            <input type="checkbox" 
                \${todo.completed ? 'checked' : ''}
                onchange="toggleTodo(\${todo.id})">
            <span class="\${todo.completed ? 'completed' : ''}">
                \${todo.text}
            </span>
            <button onclick="deleteTodo(\${todo.id})">
                Delete
            </button>
        </div>
    \`).join('');
}

// Usage
addTodo('Learn JavaScript');
addTodo('Build a project');
addTodo('Master ES6+');

// Weather Dashboard Application
class WeatherApp {
    constructor() {
        this.apiKey = 'YOUR_API_KEY';
        this.baseUrl = 'https://api.weather.com';
        this.currentCity = '';
    }
    
    async fetchWeather(city) {
        try {
            const response = await fetch(
                \`\${this.baseUrl}/weather?q=\${city}&appid=\${this.apiKey}\`
            );
            
            if (!response.ok) {
                throw new Error('City not found');
            }
            
            const data = await response.json();
            return this.formatWeatherData(data);
            
        } catch (error) {
            console.error('Weather fetch failed:', error);
            throw error;
        }
    }
    
    formatWeatherData(data) {
        return {
            city: data.name,
            temp: Math.round(data.main.temp - 273.15),
            feels: Math.round(data.main.feels_like - 273.15),
            humidity: data.main.humidity,
            description: data.weather[0].description,
            icon: data.weather[0].icon
        };
    }
    
    displayWeather(weather) {
        const container = document.getElementById('weather-display');
        
        container.innerHTML = \`
            <h2>\${weather.city}</h2>
            <div class="temp">\${weather.temp}¬∞C</div>
            <p>Feels like: \${weather.feels}¬∞C</p>
            <p>\${weather.description}</p>
            <p>Humidity: \${weather.humidity}%</p>
        \`;
    }
    
    async searchCity(city) {
        try {
            const weather = await this.fetchWeather(city);
            this.displayWeather(weather);
        } catch (error) {
            alert('Failed to fetch weather data');
        }
    }
}

// Real-time Chat Application
class ChatApp {
    constructor(username) {
        this.username = username;
        this.messages = [];
        this.users = new Map();
        this.typingUsers = new Set();
        this.eventHandlers = new Map();
    }
    
    // Event system for real-time updates
    on(event, handler) {
        if (!this.eventHandlers.has(event)) {
            this.eventHandlers.set(event, []);
        }
        this.eventHandlers.get(event).push(handler);
    }
    
    emit(event, data) {
        const handlers = this.eventHandlers.get(event) || [];
        handlers.forEach(handler => handler(data));
    }
    
    // Send message
    async sendMessage(text) {
        if (!text.trim()) return;
        
        const message = {
            id: Date.now(),
            user: this.username,
            text: text.trim(),
            timestamp: new Date(),
            delivered: false
        };
        
        // Simulate network delay
        await new Promise(resolve => setTimeout(resolve, 100));
        
        message.delivered = true;
        this.messages.push(message);
        this.emit('message', message);
        
        return message;
    }
    
    // Typing indicator
    startTyping() {
        this.typingUsers.add(this.username);
        this.emit('typing', {
            user: this.username,
            isTyping: true
        });
    }
    
    stopTyping() {
        this.typingUsers.delete(this.username);
        this.emit('typing', {
            user: this.username,
            isTyping: false
        });
    }
    
    // User management
    addUser(username, status = 'online') {
        this.users.set(username, {
            username,
            status,
            joinedAt: new Date(),
            avatar: this.generateAvatar(username)
        });
        
        this.emit('userJoined', { username, status });
    }
    
    removeUser(username) {
        this.users.delete(username);
        this.emit('userLeft', { username });
    }
    
    generateAvatar(username) {
        const colors = ['#667eea', '#764ba2', '#f093fb', '#4facfe'];
        const index = username.length % colors.length;
        return colors[index];
    }
    
    // Get formatted messages
    getMessages() {
        return this.messages.map(msg => ({
            ...msg,
            timeString: this.formatTime(msg.timestamp)
        }));
    }
    
    formatTime(date) {
        return date.toLocaleTimeString('en-US', {
            hour: '2-digit',
            minute: '2-digit'
        });
    }
}

// Chat UI Controller
class ChatUI {
    constructor(chatApp, containerId) {
        this.chat = chatApp;
        this.container = document.getElementById(containerId);
        this.setupUI();
        this.bindEvents();
    }
    
    setupUI() {
        this.container.innerHTML = \`
            <div class="chat-container">
                <div class="chat-header">
                    <h3>Chat Room</h3>
                    <span class="user-count">
                        \${this.chat.users.size} online
                    </span>
                </div>
                <div class="messages" id="messages"></div>
                <div class="typing-indicator" id="typing"></div>
                <div class="chat-input">
                    <input type="text" 
                           id="message-input" 
                           placeholder="Type a message...">
                    <button id="send-btn">Send</button>
                </div>
            </div>
        \`;
    }
    
    bindEvents() {
        const input = document.getElementById('message-input');
        const sendBtn = document.getElementById('send-btn');
        
        // Send message on click or Enter
        sendBtn.onclick = () => this.sendMessage();
        input.onkeypress = (e) => {
            if (e.key === 'Enter') this.sendMessage();
        };
        
        // Typing indicator
        let typingTimeout;
        input.oninput = () => {
            this.chat.startTyping();
            clearTimeout(typingTimeout);
            typingTimeout = setTimeout(() => {
                this.chat.stopTyping();
            }, 1000);
        };
        
        // Listen to chat events
        this.chat.on('message', (msg) => this.renderMessage(msg));
        this.chat.on('typing', (data) => this.updateTyping(data));
    }
    
    async sendMessage() {
        const input = document.getElementById('message-input');
        const text = input.value;
        
        if (text.trim()) {
            await this.chat.sendMessage(text);
            input.value = '';
            this.chat.stopTyping();
        }
    }
    
    renderMessage(message) {
        const messagesDiv = document.getElementById('messages');
        const isOwn = message.user === this.chat.username;
        
        const msgEl = document.createElement('div');
        msgEl.className = \`message \${isOwn ? 'own' : 'other'}\`;
        msgEl.innerHTML = \`
            <div class="message-user">\${message.user}</div>
            <div class="message-text">\${message.text}</div>
            <div class="message-time">\${message.timeString}</div>
        \`;
        
        messagesDiv.appendChild(msgEl);
        messagesDiv.scrollTop = messagesDiv.scrollHeight;
    }
    
    updateTyping(data) {
        const typingDiv = document.getElementById('typing');
        if (data.isTyping && data.user !== this.chat.username) {
            typingDiv.textContent = \`\${data.user} is typing...\`;
        } else {
            typingDiv.textContent = '';
        }
    }
}

// Memory Card Game
class MemoryGame {
    constructor(symbols) {
        this.symbols = [...symbols, ...symbols];
        this.cards = [];
        this.flippedCards = [];
        this.matchedPairs = 0;
        this.moves = 0;
        this.score = 0;
        this.init();
    }
    
    init() {
        this.shuffleCards();
        this.createBoard();
    }
    
    shuffleCards() {
        for (let i = this.symbols.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [this.symbols[i], this.symbols[j]] = 
                [this.symbols[j], this.symbols[i]];
        }
        
        this.cards = this.symbols.map((symbol, index) => ({
            id: index,
            symbol,
            flipped: false,
            matched: false
        }));
    }
    
    flipCard(cardId) {
        const card = this.cards.find(c => c.id === cardId);
        
        if (!card || card.flipped || card.matched) return;
        if (this.flippedCards.length >= 2) return;
        
        card.flipped = true;
        this.flippedCards.push(card);
        
        if (this.flippedCards.length === 2) {
            this.moves++;
            this.checkMatch();
        }
    }
    
    checkMatch() {
        const [card1, card2] = this.flippedCards;
        
        if (card1.symbol === card2.symbol) {
            card1.matched = true;
            card2.matched = true;
            this.matchedPairs++;
            this.score += 100;
            this.flippedCards = [];
            
            if (this.matchedPairs === this.symbols.length / 2) {
                setTimeout(() => alert('You Won!'), 300);
            }
        } else {
            setTimeout(() => {
                card1.flipped = false;
                card2.flipped = false;
                this.flippedCards = [];
                this.renderBoard();
            }, 1000);
        }
    }
    
    createBoard() {
        const board = document.getElementById('game-board');
        board.innerHTML = this.cards.map(card => \`
            <div class="card" 
                 data-id="\${card.id}"
                 onclick="game.flipCard(\${card.id})">
                <div class="card-inner">
                    <div class="card-front">?</div>
                    <div class="card-back">\${card.symbol}</div>
                </div>
            </div>
        \`).join('');
    }
    
    renderBoard() {
        this.cards.forEach(card => {
            const el = document.querySelector(
                \`[data-id="\${card.id}"]\`
            );
            el.classList.toggle('flipped', card.flipped);
            el.classList.toggle('matched', card.matched);
        });
    }
}
